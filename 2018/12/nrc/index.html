> library(nrc)

Rのコードをコンパイルできる

> compile("1+2")
.intel_syntax noprefix

  push 1

  pop rax

  pop rax

コンパイルしてアセンブル

> compile("1+2") %>% assemble %>% execute

コンパイルの流れ

compile <- function(s) {
    s %>%
        tokenize %>%
        parse %>%
        generate
}

tokenize

単語に分割する

正規表現を使って雑に実装

#' @import stringr

tokenize <- function(s) {
    s %>%
        str_replace_all("(<-|==|!=|[()+\\-*/=;])",
                        " \\1 ") %>%
        str_trim %>%
        str_split("\\s+")
}
> tokenize("a=2;a+40")
[[1]]
[1] "a"  "="  "2"  ";"  "a"  "+"  "40"
